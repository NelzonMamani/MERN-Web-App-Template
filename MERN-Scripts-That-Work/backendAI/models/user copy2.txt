const mongoose = require("mongoose");
const ROLE = {
  ADMIN: 'admin',
  BASIC: 'basic'
}

const UserSchema = new mongoose.Schema({
  name: {
    type: String,
    required: true
  },
  email: {
    type: String,
    required: true,
    unique: true
  },
  password: {
    type: String,
    required: true
  },
  role: {
    type: String,
    default: ROLE.BASIC
  }
});

module.exports = User = mongoose.model("user", UserSchema);
